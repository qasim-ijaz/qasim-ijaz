%YAML 1.2
---
$id: http://devicetree.org/schemas/clock/adi,sc5xx-cdu.yaml#
$schema: http://devicetree.org/meta-schemas/core.yaml#

title: Analog Devices SC5xx Clock Distribution Unit

maintainers:
  - Qasim Ijaz <qasim.ijaz@analog.com>

description: |
  The Clock Distribution Unit (CDU) selects and gates clocks generated by
  the Clock Generation Units (CGUs) and routes them to processor cores,
  memory interfaces, and peripheral clock domains.

properties:
  compatible:
    enum:
      - adi,sc573-cdu
      - adi,sc589-cdu
      - adi,sc594-cdu
      - adi,sc598-cdu

  reg:
    maxItems: 1

  clocks:
    description: |
      Input clocks to the CDU. Each entry corresponds to the entry at the
      same index in clock-names. The exact set depends on the compatible.
    minItems: 15
    maxItems: 20

  clock-names:
    description:
      Names of the CDU input clocks.
    minItems: 15
    maxItems: 20

  '#clock-cells':
    const: 1

allOf:
  - if:
      properties:
        compatible:
          contains:
            const: adi,sc573-cdu
    then:
      properties:
        clocks:
          minItems: 15
          maxItems: 15
        clock-names:
          items:
            - const: sys_clkin0
            - const: sys_clkin1
            - const: cclk0_0
            - const: cclk1_0

            - const: sclk1_0
